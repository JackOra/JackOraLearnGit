### 父到子的传值



#### 在子组件绑定自定义的属性指令，属性值为父组件需要传递的数据变量，子组件注册组件内部通过props数组形式接收属性名称，在子组件模版内部直接使用属性名

```vue
<!DOCTYPE html>
<html>

	<head>
		<meta charset="UTF-8">
		<title></title>
	</head>

	<body>
		<div id="app">
			<v-parent></v-parent>
		</div>
		
		<template id="parent">
			<div>
				<h2>我是父组件的内容-----{{str}}</h2>
				<v-child :name="str" :msg="tit"></v-child>
			</div>
		</template>
		
		<template id="child">
			<div>
				<h2>我是子组件的内容-----{{str}}</h2>
				<p>接收到父组件的数据----------{{msg}}{{name}}</p>
			</div>
		</template>
		<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
		<script type="text/javascript">
			var vm = new Vue({
				el: '#app',
				data: {
					
				},
				methods: {
					
				},
				components: {
					'v-parent': {
						template: '#parent',
						data:function(){
							return{
								str:'我是父组件的数据',
								tit: '你好',
							}
						},
						components: {
							'v-child': {
								props: ['name', 'msg'],
								template: '#child',
								data:function() {
									return {
										str: '我是子组件的数据'
									}
								},
							},
						}
					}
				},
			})
		</script>
	</body>

</html>
```



### 子到父的传值

#### 在子组件绑定自定义事件组件，子组件内部通过$emit('事件名', 要传递的数据)，父组件通过methods声明

```vue
<!DOCTYPE html>
<html>

	<head>
		<meta charset="UTF-8">
		<title></title>
	</head>

	<body>
		<div id="app">
			<v-parent></v-parent>
		</div>
		
		<template id="parent">
			<div>
				<h2>我是父组件</h2>
				<p>{{str}}</p>
				<p>接收子组件传递的数据---------{{str1}}</p>
				<hr />
				<v-child @toparent="getdata"></v-child>
			</div>
		</template>
		
		<template id="child">
			<div>
				<h2>我是子组件</h2>
				<p>{{str}}</p>
				<button @click="send()">发送给父组件</button>
			</div>
		</template>
		<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
		<script type="text/javascript">
			var vm = new Vue({
				
				el: '#app',
				data: {
					
				},
				methods: {
					
				},
				components: {
					'v-parent': {
						template: '#parent',
						data:function(){
							return {
								str: '我是父组件的内容',
								str1:'',
							}
						},
						methods:{
							getdata(msg){
//								console.log(msg);
								this.str1 = msg;
							},
						},
						components: {
							'v-child': {
								template: '#child',
								data: function(){
									return {
										str:'我是子组件的内容'
									}
								},
								methods: {
									send(){
										this.$emit('toparent', this.str)
									},
								},
							}
						}
					}
				},
			})
		</script>

	</body>

</html>
```
