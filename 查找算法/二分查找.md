# 二分查找

```python
#二分查找
def Binary_Search(list, key):
    low = 0
    high = len(list) - 1
    while low <= high:
        mid = (low + high) // 2
        if key == list[mid]:
            return mid
        if key > list[mid]:
            low = mid + 1
        else:
            high = mid - 1
    return None
my_list = [1,3,4,5,7,9]
print(Binary_Search(my_list,3))
```
