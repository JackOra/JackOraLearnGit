# 选择排序

```python
#先找到列表中最小元素的索引

def find_smallest(arr):
  smallest_val = arr[0]
  smallest_index = 0
  for i in range(len(arr)):
    if arr[i] < smallest_val:
      smallest_val = arr[i]
      smallest_index = i
  return smallest_index

def select_sort(arr):
  newArr = []
  for i in range(len(arr)):
    smallest_index = find_smallest(arr)
    newArr.append(arr.pop(smallest_index))
  return newArr
```
