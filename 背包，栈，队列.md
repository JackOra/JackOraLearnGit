# 背包

![test](https://github.com/JackOra/JackOraLearnGit/raw/Users/apple/Desktop/2.png)

> 可以想象一个非常喜欢收集弹子球的人。他将所有的弹子球都放在一个背包里，一次一个，并且会不时在所有的弹子球中寻找某一颗拥有某种特点的弹子球。

背包是一种***不支持从中删除元素的集合数据类型***——它的目的就是帮助用例收集元素并迭代遍历所有收集到的元素（用例也可以检查背包是否为空或者获取背包中元素的数量）。

```java
package test;

import edu.princeton.cs.introcs.StdIn;
import edu.princeton.cs.introcs.StdOut;

import java.util.Iterator;
public class Bag<Item> implements Iterable<Item>{

    private Node First;
    private int N;
    private class Node{
        Item item;
        Node next;
    }
    public boolean isEmpty(){return N==0;}
    public int size(){return N;}
    //Bag只有add方法 并且是后进先出的
    public void add(Item item){
        Node oldFirst = First;
        First = new Node();
        First.item = item;
        oldFirst.next = First;
    }
    //迭代
    public Iterator<Item> iterator(){
        return new ListIterator();
    }

    private class ListIterator implements Iterator<Item>{
        private Node Current = First; //记录链表的当前节点
        public boolean hasNext(){  //检查current是否为null
            return Current == null;
        }
        public void remove(){}
        public Item next(){ //保存当前节点的引用，将Current变量指向下一个节点并返回所保存的引用。
            Item item = Current.item;
            Current = Current.next;
            return item;
        }
    }

    public static void main(String[] args)
    {
        Bag<Double> numbers = new Bag<Double>();
        while (!StdIn.isEmpty())
            numbers.add(StdIn.readDouble());
        int N = numbers.size();
        double sum = 0.0;
        for (double x : numbers)
            sum += x;
        double mean = sum/N;
        sum = 0.0;
        for (double x : numbers)
            sum += (x - mean)*(x - mean);
        double std = Math.sqrt(sum/(N-1));
        StdOut.printf("Mean: %.2f\n", mean);
        StdOut.printf("Std dev: %.2f\n", std);
    }

}

```

## 链表

> 链表是一种递归的数据结构，它或者为空（null），或者是指向一个结点（node）的引用，该结点含有一个泛型的元素和一个指向另一条链表的引用。

### 链表的操作

- 在表头插入节点。
- 在表头删除节点。
- 在表尾插入节点。

### 链栈的实现

```java
package test;


import edu.princeton.cs.introcs.StdIn;
import edu.princeton.cs.introcs.StdOut;

import java.util.Iterator;
import java.util.ListIterator;

public class Stack<Item> implements Iterable<Item> {
    private Node First;
    private int N;

    private class Node {
        Item item;
        Node next;
    }

    public boolean isEmpty() {
        return N == 0;
    }

    public int size() {
        return N;
    }

    public void push(Item item) { //从表头添加元素
        Node oldFirst = First;
        First = new Node();
        First.item = item;
        First.next = oldFirst;
        N++;
    }

    public Item pop() { //从表头删除元素
        Item item = First.item;
        First = First.next;
        N--;
        return item;
    }

    public Iterator<Item> iterator() {
        return new ListIterator();
    }

    private class ListIterator implements Iterator<Item> {
        private Node Current = First; //记录链表当前的节点

        public boolean hasNext() {
            return Current == null; //检查Current是否为null
        }

        public void remove() {
        }

        public Item next() { 
            Item item = Current.item; //保存当前元素的引用。
            Current = Current.next; //将Current变量指向链表中下一个节点
            return item; //并且返回已经保存的引用。
        }

    }

    public static void main(String[] args) {
        Stack<String> stack = new Stack<String>();
        while (!StdIn.isEmpty()) {
            String s = StdIn.readString();
            if (!s.equals("-")) {
                stack.push(s);
            } else if (!stack.isEmpty()) {
                StdOut.println(stack.pop() + " ");
            }
            StdOut.println("(" + stack.size() + " left on stack");
        }

    }
}

```

```java
package test;

import edu.princeton.cs.introcs.StdIn;
import edu.princeton.cs.introcs.StdOut;

import java.io.FileReader;
import java.util.Iterator;

public class Quene<Item> implements Iterable<Item> {
    private Node First;
    private Node Last;
    private int N;


    private class Node{
        Item item;
        Node next;
    }
    public boolean isEmpty(){return N==0;}
    public int size(){return N;}

    //从表尾添加元素（队尾添加元素）
    public void enqueue(Item item){
        Node oldLast = Last;
        Last = new Node();
        Last.item = item;
        Last.next = null;
        if (isEmpty()){
            First = Last; //当队列中没有元素的时候First 和 Last指向同一个元素
        }else {
            oldLast.next = Last;
        }
        N++;
    }
    //从表头删除元素（队头删除元素）
    public Item dequeue(){
        Item item = First.item;
        First = First.next;
        if (isEmpty()){
            Last = null; //队列最后一个元素出来后，队列已经没有元素了
        }                //，但是Last仍然持有原来的引用，所以Last=null
        N--;
        return item;
    }

    public Iterator<Item> iterator(){
        return new ListIterator() ;
    }
    private class ListIterator implements Iterator<Item>{
        private Node Current = First; //记录当前链表的节点
        public boolean hasNext(){
            return Current == null; //检查Current是否为null
        }
        public void remove(){}
        public Item next(){
            Item item = Current.item; //保存当前节点的引用
            Current = Current.next;   //将Current指向链表的下一个节点
            return item; //返回已经保存的元素引用
        }
    }

    public static void main(String[] args) {
        Quene<String> q = new Quene<String>();
        while (!StdIn.isEmpty()){
            String s = StdIn.readString();
            if (!s.equals("-"))
                q.enqueue(s);
            else if (!q.isEmpty()) StdOut.print(q.dequeue() + " ");
        }
        StdOut.println("(" + q.size() + " left on queue");
    }
}
```
